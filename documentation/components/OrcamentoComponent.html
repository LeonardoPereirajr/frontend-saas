import { Component } from '@angular/core';
import { MatTableDataSource } from '@angular/material/table';
import jsPDF from 'jspdf';
import autoTable from 'jspdf-autotable';

interface OrcamentoItem {
  servico: string;
  preco: number;
  quantidade: number;
}

@Component({
  selector: 'app-orcamento',
  templateUrl: './orcamento.component.html',
  styleUrls: ['./orcamento.component.css']
})
export class OrcamentoComponent {
  displayedColumns: string[] = ['servico', 'preco', 'quantidade', 'total', 'acoes'];
  dataSource = new MatTableDataSource<OrcamentoItem>([]);
  
  totalGeral = 0;

  constructor() {
    this.adicionarItem();
  }

  adicionarItem() {
    this.dataSource.data = [...this.dataSource.data, { servico: '', preco: 0, quantidade: 1 }];
    this.calcularTotal();
  }

  calcularTotal() {
    this.totalGeral = this.dataSource.data.reduce((acc, item) => acc + (item.preco * item.quantidade), 0);
  }
}
